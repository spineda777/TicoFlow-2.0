                <button className="btn-ghost" style={{fontSize:11,padding:"4px 10px"}} onClick={()=>{if(editNotes){onUpdate({...c,notes});toast("✓ Notas guardadas");}setEditNotes(!editNotes);}}>{editNotes?"✓ Guardar":"✎ Editar"}</button>
              </div>
              {editNotes?<textarea className="input-field" rows={8} value={notes} onChange={e=>setNotes(e.target.value)} style={{resize:"vertical"}} placeholder="Preferencias, alergias, historial..."/>:<div style={{padding:"14px 16px",background:"rgba(29,175,142,0.04)",borderRadius:10,border:"1px solid rgba(29,175,142,0.08)",fontSize:13,color:notes?"#CFD8DC":"#37474F",lineHeight:1.7,minHeight:80}}>{notes||"Sin notas. Haz clic en Editar para añadir."}</div>}
            </div>
          )}
        </div>
      </div>
    </>
  );
}
